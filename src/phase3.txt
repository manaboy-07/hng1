# 🔐 STEP 3 — Credentials Storage

Path:

```txt
~/.insighta/credentials.json
```

---

## Utility file

```ts
import * as fs from 'fs-extra';
import * as path from 'path';
import * as os from 'os';

const dir = path.join(os.homedir(), '.insighta');
const file = path.join(dir, 'credentials.json');

export async function saveCredentials(data) {
  await fs.ensureDir(dir);
  await fs.writeJson(file, data);
}

export async function getCredentials() {
  if (!(await fs.pathExists(file))) return null;
  return fs.readJson(file);
}

export async function clearCredentials() {
  await fs.remove(file);
}
```

# 🔁 STEP 5 — Token Auto Refresh (IMPORTANT)

Create request wrapper:

```ts
async function apiRequest(config) {
  let creds = await getCredentials();

  try {
    return await axios({
      ...config,
      headers: {
        Authorization: `Bearer ${creds.access_token}`,
        'X-API-Version': '1',
      },
    });
  } catch (err) {
    if (err.response?.status === 401) {
      // 🔥 refresh token
      const res = await axios.post('/auth/refresh', {
        refresh_token: creds.refresh_token,
      });

      await saveCredentials(res.data);

      // retry original request
      return axios({
        ...config,
        headers: {
          Authorization: `Bearer ${res.data.access_token}`,
        },
      });
    }

    throw err;
  }
}
```

```ts
program

    const res = await apiRequest({
      url: '/profiles',
      params: options,
    });

    displayTable(res.data.data);
  });
```

# 📋 STEP 7 — Table Output

```ts
import Table from 'cli-table3';

function displayTable(data) {
  const table = new Table({
    head: ['ID', 'Name', 'Gender', 'Age'],
  });

  data.forEach((row) => {
    table.push([row.id, row.name, row.gender, row.age]);
  });

  console.log(table.toString());
}
```
